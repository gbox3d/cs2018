#include "stdafx.h"


int main()
{
	FILE *fp;
	char ch;

	fopen_s(&fp,"a.txt","r");
	if (fp == NULL) {
		printf_s("파일읽기실패 \n");
	}
	else {

		while (1) {
			ch = fgetc(fp);
			if (ch == EOF) {
				break;
			}
			else {
				printf_s("%c",ch);
			}
		}
		printf_s("\n");
		/*
		ch = fgetc(fp);
		//ch = fgetc(stdin);
		printf_s("입력한 문자 : %c \n", ch);
		ch = fgetc(fp);
		//ch = fgetc(stdin);
		printf_s("입력한 문자 : %c \n", ch);
		*/
		fclose(fp);
	}

    return 0;
}
